country_event = {
	id = neu_advisor.4
	title = neu_advisor.4.t
	desc = neu_advisor.4.d
	
	trigger = {
		has_idea_with_trait = staunch_traditionalist
		NOT = { has_government = neutrality }
		neutrality > 0.25
	}

	mean_time_to_happen = {
		days = 365
		modifier = {
			factor = 0.75
			has_country_flag = neutrality_popular_movement
		}
	}
	
	option = {
		name = neu_advisor.4.a
		if = {
			limit = {
				has_government = democratic
			}
			if = {
				limit = { neutrality > 0.4 }
				start_civil_war = {
					ruling_party = neutrality
					ideology = democratic
					size = 0.6
				}
			}
			if = {
				limit = { NOT = { neutrality > 0.4 } }
				start_civil_war = {
					ruling_party = neutrality
					ideology = democratic
					size = 0.7
				}
			}
		}
		if = {
			limit = {
				has_government = communism
			}
			if = {
				limit = { neutrality > 0.4 }
				start_civil_war = {
					ruling_party = neutrality
					ideology = communism
					size = 0.6
				}
			}
			if = {
				limit = { NOT = { neutrality > 0.4 } }
				start_civil_war = {
					ruling_party = neutrality
					ideology = communism
					size = 0.7
				}
			}
		}
		if = {
			limit = {
				has_government = fascism
			}
			if = {
				limit = { neutrality > 0.4 }
				start_civil_war = {
					ruling_party = neutrality
					ideology = fascism
					size = 0.6
				}
			}
			if = {
				limit = { NOT = { neutrality > 0.4 } }
				start_civil_war = {
					ruling_party = neutrality
					ideology = fascism
					size = 0.7
				}
			}
		}
	}
	option = {
		name = neu_advisor.4.b
	}
}
